消息提醒场景

内容提醒框

内容提醒框
示例:
bui.confirm("消息提醒",function (e){ console.log("你点击了"+$(this).text()) });

带图标提醒

点击打开成功提醒
示例:
bui.confirm({ "title": "", "content":'<div class="bui-box-center"><h3><i class="icon-success"></i></h3><p>提交成功</p></div>', "buttons":[{name:"确定",className:"primary-reverse"}] });

带标题居中提醒

点击打开标题提醒
示例:
bui.confirm({ "title": "", "content":'<div class="bui-box-center"><h3><i class="icon-errorfill"></i>温馨提醒</h3><p>提交成功</p></div>', "buttons":[{name:"取消",className:""},{name:"确定",className:"primary-reverse"}] });
