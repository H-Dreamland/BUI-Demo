列表侧滑场景

  • bui.listview+bui.dialog示例
示例:

html:

// 注意: listview 的结构跟list 是不一样的 <ul id="listview" class="bui-listview"> <li> <div class="bui-btn bui-box"> <div class="span1">左滑动出现菜单</div> <i class="icon-listright"></i> </div> </li> </ul>

js:

var listView = bui.listview({ id: "#listview", data: [{ "text": "修改", "classname":"primary" },{ "text": "删除", "classname":"danger" }], callback: function (e,menu) { //关闭菜单 menu.close(); } });
