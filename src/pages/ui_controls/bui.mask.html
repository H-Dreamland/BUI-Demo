// 默认是自动触发, 这样可以只实例化一个 appendTo 为空则添加在整个页面 var uiMask = bui.mask({ appendTo: "#mask", autoTrigger: false }); //添加 $('#add').on("click",function (argument) { uiMask.add(); }) //移除 $('#remove').on("click",function (argument) { uiMask.remove(); }) //显示 $('#show').on("click",function (argument) { uiMask.show(); }) //隐藏 $('#hide').on("click",function (argument) { uiMask.hide(); })
