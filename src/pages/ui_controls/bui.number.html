数字条

动态绑定


设置值
获取值
示例:

html:

<div id="number" class="bui-number"></div>

js:

var uiNumber = bui.number({ id: '#number', value: 5, min: 1, max: 10 })

静态绑定

示例:

html:

<div id="number2" class="bui-number"> <div class="bui-number-prev"> <i class="icon-minus"></i> </div> <input type="text" value="2"/> <div class="bui-number-next"> <i class="icon-plus"></i> </div> </div>

js:

var uiNumber2 = bui.number({ id: '#number2', value: 5, min: 1, max: 10, render: false, })
